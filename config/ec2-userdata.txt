#!/bin/bash
wget https://github.com/purnaresa/dum-dum-app/releases/download/beta/run-linux-amd64 -O dumdumapp
chmod +x dumdumapp
sudo mkdir /root/go
sudo mv dumdumapp /root/go/
wget https://github.com/purnaresa/dum-dum-app/releases/download/0.2-beta/dum-dum-app.service -O dumdumapp.service
sudo mv dumdumapp.service /etc/systemd/system/
sudo systemctl start dumdumapp.service
sudo systemctl enable dumdumapp.service
